# ========================================
# TEKNOLOJI MENAJERI - Environment Variables
# ========================================

# ========================================
# 1. FRONTEND ENVIRONMENT VARIABLES
# ========================================
# Dosya: .env (proje root dizini)

VITE_SUPABASE_URL=https://YOUR_PROJECT_REF.supabase.co
VITE_SUPABASE_ANON_KEY=YOUR_ANON_KEY_HERE
VITE_RECAPTCHA_SITE_KEY=YOUR_RECAPTCHA_SITE_KEY

# Nasıl alınır:
# - Supabase Dashboard > Settings > API
# - reCAPTCHA: https://www.google.com/recaptcha/admin


# ========================================
# 2. SUPABASE EDGE FUNCTIONS SECRETS
# ========================================
# Komut: supabase secrets set KEY=VALUE

# 2.1 Gotenberg PDF Service
GOTENBERG_URL=http://host.docker.internal:3000
# Local: http://localhost:3000
# Docker: http://host.docker.internal:3000
# Production: http://your-gotenberg-server:3000

# 2.2 Mailgun Email Service
MAILGUN_API_KEY=YOUR_MAILGUN_API_KEY
MAILGUN_DOMAIN=YOUR_MAILGUN_DOMAIN
# Nasıl alınır:
# - Mailgun Dashboard > Settings > API Keys
# - Domain: Settings > Domains

# 2.3 Google APIs
GOOGLE_PAGESPEED_API_KEY=YOUR_PAGESPEED_API_KEY
GEMINI_API_KEY=YOUR_GEMINI_API_KEY
# Nasıl alınır:
# - PageSpeed: https://console.cloud.google.com > APIs & Services
# - Gemini: https://makersuite.google.com/app/apikey

# 2.4 GitHub (Opsiyonel - Markdown Backup)
GITHUB_TOKEN=YOUR_GITHUB_TOKEN
# Nasıl alınır:
# - https://github.com/settings/tokens
# - Permission: gist (Create gists)

# 2.5 Supabase Internal
SUPABASE_URL=https://YOUR_PROJECT_REF.supabase.co
SUPABASE_SERVICE_ROLE_KEY=YOUR_SERVICE_ROLE_KEY
# Nasıl alınır:
# - Supabase Dashboard > Settings > API
# - Service Role Key (SECRET - sakla!)


# ========================================
# 3. KOMUTLARLA SECRETS EKLEME
# ========================================

# Önce Supabase'e login olun
# supabase login
# supabase link --project-ref YOUR_PROJECT_REF

# Sonra secrets'ları ekleyin:
# supabase secrets set GOTENBERG_URL=http://host.docker.internal:3000
# supabase secrets set MAILGUN_API_KEY=YOUR_KEY
# supabase secrets set MAILGUN_DOMAIN=YOUR_DOMAIN
# supabase secrets set GOOGLE_PAGESPEED_API_KEY=YOUR_KEY
# supabase secrets set GEMINI_API_KEY=YOUR_KEY
# supabase secrets set GITHUB_TOKEN=YOUR_TOKEN
# supabase secrets set SUPABASE_URL=https://YOUR_PROJECT_REF.supabase.co
# supabase secrets set SUPABASE_SERVICE_ROLE_KEY=YOUR_KEY

# Kontrol edin:
# supabase secrets list


# ========================================
# 4. GÜVENLİK NOTLARI
# ========================================

# ❌ ASLA YAPMAYINIZ:
# - .env dosyasını Git'e commit etmeyin
# - API key'leri public repository'de paylaşmayın
# - Service Role Key'i frontend'de kullanmayın

# ✅ YAPINIZ:
# - .env dosyasını .gitignore'a ekleyin (zaten eklendi)
# - Production'da environment variables kullanın
# - Key rotation yapın (düzenli olarak değiştirin)
# - Sadece gerekli permissions verin


# ========================================
# 5. PRODUCTION DEPLOYMENT
# ========================================

# Vercel Deployment:
# - Vercel Dashboard > Settings > Environment Variables
# - Tüm VITE_ ile başlayan değişkenleri ekleyin

# Netlify Deployment:
# - Netlify Dashboard > Site Settings > Environment Variables
# - Tüm VITE_ ile başlayan değişkenleri ekleyin


# ========================================
# 6. NOTLAR
# ========================================

# Local Development:
# - .env dosyası kullanın
# - Docker localhost:3000
# - npm run dev

# Production:
# - Platform environment variables kullanın
# - Gotenberg remote server veya Cloud Run
# - npm run build && npm run preview
